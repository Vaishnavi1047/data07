#PCA

from sklearn.decomposition import PCA
from sklearn.preprocessing import StandardScaler
import pandas as pd
import matplotlib.pyplot as plt
# 1. Import and prepare data
# 2. Apply one-hot encoding to df_reduced
df_encoded = pd.get_dummies(df_reduced, drop_first=True)

# 3. Define feature matrix X and target vector y
# Assuming 'survived' is the target column
X = df_encoded.drop(columns=['survived'])
y = df_encoded['survived']
# 4. Standardize the features
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)
# 5. Apply PCA (reduce to 7 components)
pca = PCA(n_components=7)
X_pca = pca.fit_transform(X_scaled)

# 6. Create df_pca with components PC1 through PC7
df_pca = pd.DataFrame(
X_pca,
columns=[f'PC{i+1}' for i in range(7)]
)
# 7. Visualize PC1 vs PC2
plt.figure(figsize=(10, 6))
plt.scatter(df_pca['PC1'], df_pca['PC2'], c=y, cmap='coolwarm', alpha=0.7)
plt.xlabel('PC1')
plt.ylabel('PC2')
plt.title('PCA: PC1 vs PC2 (Colored by Survival)')
plt.colorbar(label='Survived')
plt.grid(True)
plt.show()
# 8. Print Explained Variance Ratio and Component Loadings
print("\nExplained Variance Ratio for 7 Components:")
for i, ratio in enumerate(pca.explained_variance_ratio_):
	print(f"PC{i+1}: {ratio:.4f}")
print("\nPCA Component Loadings:")
loadings = pd.DataFrame(
	pca.components_.T,
	columns=[f'PC{i+1}' for i in range(7)],
	index=X.columns
)
display(loadings)
