import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.datasets import load_iris

# 1. Load Iris dataset and filter for Class 0 (setosa) and Class 1 (versicolor)
iris = load_iris()
df = pd.DataFrame(data=iris.data, columns=iris.feature_names)
df['target'] = iris.target

# Filter to keep only classes 0 and 1
df_filtered = df[df['target'].isin([0, 1])]
print(f"Shape of filtered dataset: {df_filtered.shape}")

# 2. Compute mean (μ) and standard deviation (σ) for each feature per class
stats = df_filtered.groupby('target').agg(['mean', 'std']).T
print("\nMean and Std Dev table for all 4 features:")
print(stats)

# 3. Using the formula to compute Fisher Score for each feature
# Formula: Fisher Score = (μ1 - μ2)² / (σ1² + σ2²)
fisher_scores = {}

for feature in iris.feature_names:
    mu0 = stats.loc[(feature, 'mean'), 0]
    mu1 = stats.loc[(feature, 'mean'), 1]
    sigma0 = stats.loc[(feature, 'std'), 0]
    sigma1 = stats.loc[(feature, 'std'), 1]
    
    score = ((mu0 - mu1)**2) / (sigma0**2 + sigma1**2)
    fisher_scores[feature] = score

fisher_series = pd.Series(fisher_scores)
print("\nFisher Scores:")
print(fisher_series)

# 4. Sort the features in descending order
ranked_features = fisher_series.sort_values(ascending=False)
print("\nRanked List of Features:")
print(ranked_features)

best_feature = ranked_features.index[0]
worst_feature = ranked_features.index[-1]
print(f"\nBest feature: {best_feature}")
print(f"Worst feature: {worst_feature}")

# 5. Plot a bar chart and explanation
plt.figure(figsize=(10, 6))
ranked_features.plot(kind='bar', color='skyblue', edgecolor='black')
plt.title('Fisher Scores for Iris Features (Setosa vs Versicolor)')
plt.xlabel('Features')
plt.ylabel('Fisher Score')
plt.xticks(rotation=45)
plt.grid(axis='y', linestyle='--', alpha=0.7)
plt.show()

# Conclusion
print("\nObservation:")
print("A high Fisher Score indicates that the feature has a large difference between class means ")
print("and low intra-class variance, making it an excellent discriminator for classification.")


#DATA TRANSFORMATION USING TITANIC
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.preprocessing import MinMaxScaler

# 1. Load the dataset and handle missing values
# Note: Ensure 'titanic.csv' is in your directory, or use seaborn's load_dataset
import seaborn as sns
df = sns.load_dataset('titanic')

# Fill Age with median and Embarked with mode
df['age'] = df['age'].fillna(df['age'].median())
df['embarked'] = df['embarked'].fillna(df['embarked'].mode()[0])

# Drop Cabin column
df.drop(columns=['deck'], inplace=True, errors='ignore') # 'deck' is the cabin column in seaborn's version
print("1. Missing values handled and Cabin column dropped.")

# 2. Convert Sex to numeric (male = 1, female = 0)
df['sex'] = df['sex'].map({'male': 1, 'female': 0})
print("\n2. Sex column encoded. First 5 rows:")
print(df[['sex', 'age', 'fare']].head())

# 3. Min-Max Normalization for Age and Fare
scaler = MinMaxScaler()
df[['age', 'fare']] = scaler.fit_transform(df[['age', 'fare']])

print("\n3. Normalization Results:")
print(f"Age - Min: {df['age'].min()}, Max: {df['age'].max()}")
print(f"Fare - Min: {df['fare'].min()}, Max: {df['fare'].max()}")

# 4. Create age_group by binning
bins = [0, 17, 59, 100]
labels = ['Child', 'Adult', 'Senior']
# Re-loading original age for binning since Q4 asks to bin "original age"
original_df = sns.load_dataset('titanic')
df['age_group'] = pd.cut(original_df['age'].fillna(original_df['age'].median()), 
                         bins=bins, labels=labels)

print("\n4. Value counts for age_group:")
print(df['age_group'].value_counts())

# 5. Plot Boxplots for Fare before and after normalization
fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(12, 5))

# Before normalization (using original data for plot)
sns.boxplot(y=original_df['fare'], ax=ax1, color='salmon')
ax1.set_title('Fare Before Normalization')

# After normalization
sns.boxplot(y=df['fare'], ax=ax2, color='skyblue')
ax2.set_title('Fare After Normalization (0-1 Scale)')

plt.tight_layout()
plt.show()

# Two-line observation
print("\n5. Observation:")
print("The boxplot shows that while the scale of the Y-axis changed from actual currency to a 0-1 range,")
print("the relative distribution and the presence of outliers remain identical after normalization.")

#NAIV BAYES
import kagglehub
from kagglehub import KaggleDatasetAdapter
import pandas as pd
from sklearn.preprocessing import LabelEncoder
from sklearn.naive_bayes import CategoricalNB
from sklearn.metrics import classification_report

# --- 1. Load the dataset ---
file_path = "Buy_Computer.csv"
df = kagglehub.load_dataset(
    KaggleDatasetAdapter.PANDAS,
    "usman27/buy-computer",
    file_path,
)

print("Dataset Shape:", df.shape)
# Checking for 'buys_computer' column (or similar name in the CSV)
target_col = 'buys_computer' 
print("\nClass counts:\n", df[target_col].value_counts())

# --- 2. Encode categorical columns ---
# We create a dictionary to store encoders for each column so we can transform test data later
encoders = {}
encoded_df = df.copy()

for col in df.columns:
    le = LabelEncoder()
    encoded_df[col] = le.fit_transform(df[col])
    encoders[col] = le

print("\nEncoded Data (First 5 rows):\n", encoded_df.head())

# --- 3. Split features and target, and train CategoricalNB ---
X = encoded_df.drop(target_col, axis=1)
y = encoded_df[target_col]

model = CategoricalNB()
model.fit(X, y)
print(f"\nTraining Accuracy: {model.score(X, y):.4f}")

# --- 4. Predict for new customer ---
# Requirement: age=youth, income=medium, student=yes, credit_rating=fair
# We use the saved encoders to transform these strings into the correct integers
new_customer_raw = {
    'age': 'youth',
    'income': 'medium',
    'student': 'yes',
    'credit_rating': 'fair'
}

# Convert raw input to encoded list based on the training encoders
new_customer_encoded = []
for col in X.columns:
    val = new_customer_raw[col]
    encoded_val = encoders[col].transform([val])[0]
    new_customer_encoded.append(encoded_val)

prediction = model.predict([new_customer_encoded])
predicted_label = encoders[target_col].inverse_transform(prediction)[0]
print(f"\nPredicted class for new customer: {predicted_label}")

# --- 5. Classification Report ---
y_pred = model.predict(X)
target_names = encoders[target_col].classes_.astype(str)
report = classification_report(y, y_pred, target_names=target_names)

print("\nClassification Report:\n", report)

# Comparison logic
f1_scores = classification_report(y, y_pred, output_dict=True)
best_class = max(f1_scores['no']['f1-score'], f1_scores['yes']['f1-score'])
print(f"The model classifies better for the class with the higher F1-score.")
