import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.datasets import load_iris

# 1. Load Iris dataset and filter for Class 0 (setosa) and Class 1 (versicolor)
iris = load_iris()
df = pd.DataFrame(data=iris.data, columns=iris.feature_names)
df['target'] = iris.target

# Filter to keep only classes 0 and 1
df_filtered = df[df['target'].isin([0, 1])]
print(f"Shape of filtered dataset: {df_filtered.shape}")

# 2. Compute mean (μ) and standard deviation (σ) for each feature per class
stats = df_filtered.groupby('target').agg(['mean', 'std']).T
print("\nMean and Std Dev table for all 4 features:")
print(stats)

# 3. Using the formula to compute Fisher Score for each feature
# Formula: Fisher Score = (μ1 - μ2)² / (σ1² + σ2²)
fisher_scores = {}

for feature in iris.feature_names:
    mu0 = stats.loc[(feature, 'mean'), 0]
    mu1 = stats.loc[(feature, 'mean'), 1]
    sigma0 = stats.loc[(feature, 'std'), 0]
    sigma1 = stats.loc[(feature, 'std'), 1]
    
    score = ((mu0 - mu1)**2) / (sigma0**2 + sigma1**2)
    fisher_scores[feature] = score

fisher_series = pd.Series(fisher_scores)
print("\nFisher Scores:")
print(fisher_series)

# 4. Sort the features in descending order
ranked_features = fisher_series.sort_values(ascending=False)
print("\nRanked List of Features:")
print(ranked_features)

best_feature = ranked_features.index[0]
worst_feature = ranked_features.index[-1]
print(f"\nBest feature: {best_feature}")
print(f"Worst feature: {worst_feature}")

# 5. Plot a bar chart and explanation
plt.figure(figsize=(10, 6))
ranked_features.plot(kind='bar', color='skyblue', edgecolor='black')
plt.title('Fisher Scores for Iris Features (Setosa vs Versicolor)')
plt.xlabel('Features')
plt.ylabel('Fisher Score')
plt.xticks(rotation=45)
plt.grid(axis='y', linestyle='--', alpha=0.7)
plt.show()

# Conclusion
print("\nObservation:")
print("A high Fisher Score indicates that the feature has a large difference between class means ")
print("and low intra-class variance, making it an excellent discriminator for classification.")

encoders = {}
encoded_df = df.copy()

for col in df.columns:
    le = LabelEncoder()
    encoded_df[col] = le.fit_transform(df[col])
    encoders[col] = le

print("\nEncoded Data (First 5 rows):\n", encoded_df.head())

# --- 3. Split features and target, and train CategoricalNB ---
X = encoded_df.drop(target_col, axis=1)
y = encoded_df[target_col]

model = CategoricalNB()

print(f"\nTraining Accuracy: {model.score(X, y):.4f}")

# --- 4. Predict for new customer ---
# Requirement: age=youth, income=medium, student=yes, credit_rating=fair
# We use the saved encoders to transform these strings into the correct integers

    'age': 'youth',
    'income': 'medium',
    'student': 'yes',
    'credit_rating': 'fair'
}

new_customer_encoded = []

    val = new_customer_raw[col]

    new_customer_encoded.append(encoded_val)

# --- 5. Classification Report ---
y_pred = model.predict(X)

report = classification_report(y, y_pred, target_names=target_names)

# 4. Fisher Score Calculation
def calculate_fisher_score(X, y_encoded):
    features = X.columns
    fisher_scores = {}
    unique_classes = np.unique(y_encoded)

    for feature in features:
        scores_per_class = []
        for cls in unique_classes:
            X_cls = X[y_encoded == cls][feature]
            X_not_cls = X[y_encoded != cls][feature]

            if len(X_cls) > 1 and len(X_not_cls) > 1:
                mean_cls = np.mean(X_cls)
                mean_not_cls = np.mean(X_not_cls)
                var_cls = np.var(X_cls)
                var_not_cls = np.var(X_not_cls)

                if (var_cls + var_not_cls) != 0:
                    score = (mean_cls - mean_not_cls)**2 / (var_cls + var_not_cls)
                    scores_per_class.append(score)

        fisher_scores[feature] = np.mean(scores_per_class) if scores_per_class else 0

    return pd.Series(fisher_scores).sort_values(ascending=False)

fisher_scores = calculate_fisher_score(X, y_encoded)

# 1. Prepare a dictionary that counts the number of “yes” and “no” class labels for each nominal value in an attribute.
def count_class_labels_per_attribute_value(df, target_attribute, feature_attributes):
    counts = {}
    for attr in feature_attributes:
        counts[attr] = {}
        for value in df[attr].unique():
            counts[attr][value] = {
                'yes': len(df[(df[attr] == value) & (df[target_attribute] == 'yes')]),
                'no': len(df[(df[attr] == value) & (df[target_attribute] == 'no')])
            }


    return counts


# 2. Compute the probability of the class labels (buys_computer = “yes” or “no”) for each nominal value in an attribute.
def compute_probabilities(df, target_attribute, feature_attributes, counts):
    probabilities = {}

    total_instances = len(df)
    prior_yes = len(df[df[target_attribute] == 'yes']) / total_instances
    prior_no = len(df[df[target_attribute] == 'no']) / total_instances

    probabilities['prior'] = {'yes': prior_yes, 'no': prior_no}


    for attr in feature_attributes:
        probabilities[attr] = {}
        for value in df[attr].unique():
            count_yes = counts[attr][value]['yes']
            count_no = counts[attr][value]['no']



            denominator_yes = len(df[df[target_attribute] == 'yes'])
            denominator_no = len(df[df[target_attribute] == 'no'])

            prob_yes = (count_yes) / denominator_yes
            prob_no = (count_no) / denominator_no

            probabilities[attr][value] = {
                'yes': prob_yes,
                'no': prob_no
            }
    return probabilities

import pandas as pd

class UniversalNaiveBayes:
    def __init__(self):
        self.prior_probs = {}
        self.conditional_probs = {}
        self.target = None
        self.features = []

    def fit(self, df, target_column):
        """Learns probabilities from any categorical dataframe."""
        self.target = target_column
        self.features = [col for col in df.columns if col != target_column]
        total_rows = len(df)
        
        # 1. Calculate Prior Probabilities: P(Class)
        class_counts = df[target_column].value_counts()
        for label, count in class_counts.items():
            self.prior_probs[label] = count / total_rows
            
        # 2. Calculate Conditional Probabilities: P(Feature=Value | Class)
        for col in self.features:
            self.conditional_probs[col] = {}
            for val in df[col].unique():
                self.conditional_probs[col][val] = {}
                for label in class_counts.index:
                    # Count matches for specific feature value AND class label
                    match_count = len(df[(df[col] == val) & (df[target_column] == label)])
                    # Probability = Count of matches / Total instances of that class
                    self.conditional_probs[col][val][label] = match_count / class_counts[label]
        
        print(f"Model trained on {len(self.features)} features for target: '{target_column}'")

    def predict(self, instance_dict):
        """Predicts the class for a new dictionary of features."""
        results = {}
        
        for label in self.prior_probs:
            # Start with the Prior Probability
            probability = self.prior_probs[label]
            
            # Multiply by each Conditional Probability
            for col, value in instance_dict.items():
                if col in self.conditional_probs and value in self.conditional_probs[col]:
                    probability *= self.conditional_probs[col][value][label]
                else:
                    # Handle unseen values (Simple smoothing)
                    probability *= 0.000001 
            
            results[label] = probability
            
        # Return the label with the highest probability
        return max(results, key=results.get), results

# 3. Implement the perceptron update rule.
# 4. Train the model for multiple epochs until convergence.

class Perceptron:
    def __init__(self, learning_rate=0.01, n_iterations=100):
        self.learning_rate = learning_rate
        self.n_iterations = n_iterations
        self.weights = None
        self.bias = None
        self.errors_ = []

    def fit(self, X, y):
        self.weights = np.random.rand(X.shape[1]) * 0.01
        self.bias = 0.01 * np.random.randint(300000)

        for _ in range(self.n_iterations):
            errors = 0
            for xi, target in zip(X, y):
                linear_output = np.dot(xi, self.weights) + self.bias
                #yout=x*w+bias
                y_pred = np.where(linear_output >= 0.0, 1, -1)


                if y_pred != target:
                    update = self.learning_rate * (target - y_pred)
                    self.weights += update * xi
                    self.bias += update
                    errors += 1
            self.errors_.append(errors)
            if errors == 0:
                print(f"Converged after {_ + 1} iterations.")
                break
        return self

    def predict(self, X):
        linear_output = np.dot(X, self.weights) + self.bias
        return np.where(linear_output >= 0.0, 1, -1)

perceptron = Perceptron(learning_rate=0.01, n_iterations=100)

#3 Fisher Score for every attribute

def fisher_scores(X, y, class_means_dict, class_stds_dict):
    unique_classes = np.unique(y)
    num_features = X.shape[1]
    fisher_score = pd.Series(index=X.columns, dtype=float)

    overall_mean = X.mean(axis=0)

    for feature_idx, feature_name in enumerate(X.columns):
        numerator = 0
        denominator = 0

        for cls in unique_classes:
            mean_cls_feature = class_means_dict[cls][feature_name]
            std_cls_feature = class_stds_dict[cls][feature_name]

            n_cls = np.sum(y == cls)

            numerator += n_cls * (mean_cls_feature - overall_mean[feature_name])**2

            denominator += n_cls * (std_cls_feature**2)
            #denominator += n_cls * (std_cls_feature**2 + 1e-6)

        fisher_score[feature_name] = numerator / denominator

    return fisher_score

fisher_scores_result = fisher_scores(X, y, scores, class_std_devs)

import numpy as np

def calculate_entropy(target_series):
    counts = target_series.value_counts()
    probabilities = counts / len(target_series)
    entropy = -np.sum([p * np.log2(p) for p in probabilities if p > 0])

    return entropy

total_entropy = calculate_entropy(y_buy)

print(f"Total Dataset Entropy (y_buy): {total_entropy:.4f}")
#2. Compute entropy for each attribute.
def calculate_conditional_entropy(feature_series, target_series):
    unique_values = feature_series.unique()
    total_samples = len(feature_series)
    conditional_entropy = 0

    for value in unique_values:
        subset_target = target_series[feature_series == value]

        weight = len(subset_target) / total_samples

        subset_entropy = calculate_entropy(subset_target)

        conditional_entropy += weight * subset_entropy

    return conditional_entropy
