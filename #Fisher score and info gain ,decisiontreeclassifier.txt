#mean of each class
import numpy as np
import pandas as pd
def class_means(X, y):
    if isinstance(y, pd.DataFrame) and y.shape[1] == 1:
      y_series = y.iloc[:, 0]
    else:
      y_series = y
    classes = np.unique(y_series)
    mean_dict = {}
    for cls in classes:
      mean_dict[cls] = X[y_series == cls].mean(axis=0)
    return mean_dict

scores=class_means(X, y)
print(scores)

#2 Standard deviation for each class
def class_stds(X, y):
    if isinstance(y, pd.DataFrame) and y.shape[1] == 1:
      y_series = y.iloc[:, 0]
    else:
      y_series = y
    classes = np.unique(y_series)
    std_dict = {}
    for cls in classes:
      std_dict[cls] = X[y_series == cls].std(axis=0)
    return std_dict

class_std_devs = class_stds(X, y)
print("Standard Deviation of attribute variables for each class:")
print(class_std_devs)

#3 Fisher Score for every attribute

def fisher_scores(X, y, class_means_dict, class_stds_dict):
    unique_classes = np.unique(y)
    num_features = X.shape[1]
    fisher_score = pd.Series(index=X.columns, dtype=float)

    overall_mean = X.mean(axis=0)

    for feature_idx, feature_name in enumerate(X.columns):
        numerator = 0
        denominator = 0

        for cls in unique_classes:
            mean_cls_feature = class_means_dict[cls][feature_name]
            std_cls_feature = class_stds_dict[cls][feature_name]

            n_cls = np.sum(y == cls)

            numerator += n_cls * (mean_cls_feature - overall_mean[feature_name])**2

            denominator += n_cls * (std_cls_feature**2)
            #denominator += n_cls * (std_cls_feature**2 + 1e-6)

        fisher_score[feature_name] = numerator / denominator

    return fisher_score

fisher_scores_result = fisher_scores(X, y, scores, class_std_devs)
print("Fisher Scores for each feature:")
print(fisher_scores_result)

#4 Rank all attributes based on Fisher Score.

def rank_features(scores):
    ranked_indices = scores.sort_values(ascending=False).index
    return ranked_indices

ranked_indices = rank_features(fisher_scores_result)
print("Feature ranking (highest to lowest Fisher Score):")
print(ranked_indices.tolist())

#5 Select the top k features where k < n.

def select_top_k_features(X, scores, k):
    ranked_indices = scores.sort_values(ascending=False).index
    top_features = ranked_indices[:k]
    X_top_k = X[top_features]
    return X_top_k, top_features

k = 5

print(f"Top {k} feature names:", top_features.tolist())
print(f"First 5 rows of X with top {k} features:")
print(X_top_k.head())

import numpy as np

def calculate_entropy(target_series):
    counts = target_series.value_counts()
    probabilities = counts / len(target_series)
    entropy = -np.sum([p * np.log2(p) for p in probabilities if p > 0])

    return entropy

total_entropy = calculate_entropy(y_buy)

print(f"Total Dataset Entropy (y_buy): {total_entropy:.4f}")
#2. Compute entropy for each attribute.
def calculate_conditional_entropy(feature_series, target_series):
    unique_values = feature_series.unique()
    total_samples = len(feature_series)
    conditional_entropy = 0

    for value in unique_values:
        subset_target = target_series[feature_series == value]

        weight = len(subset_target) / total_samples

        subset_entropy = calculate_entropy(subset_target)

        conditional_entropy += weight * subset_entropy

    return conditional_entropy

conditional_entropies = {}
for column in X_buy.columns:
    c_entropy = calculate_conditional_entropy(X_buy[column], y_buy)
    conditional_entropies[column] = c_entropy

from sklearn.model_selection import train_test_split
from sklearn.tree import DecisionTreeClassifier, export_text
import pandas as pd

# 1. Prepare feature matrix X and target vector y

y_dt = diabetes_df['Outcome']

# 2. Split the data (80% train, 20% test)
X_train, X_test, y_train, y_test = train_test_split(X_dt, y_dt, test_size=0.2, random_state=42)

# 3. Initialize Decision Tree with max_depth=3 for readability
