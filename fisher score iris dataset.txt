fisher score iris dataset
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.datasets import load_iris

# 1. Load Iris dataset and filter for Class 0 (setosa) and Class 1 (versicolor)
iris = load_iris()
df = pd.DataFrame(data=iris.data, columns=iris.feature_names)
df['target'] = iris.target

# Filter to keep only classes 0 and 1
df_filtered = df[df['target'].isin([0, 1])]
print(f"Shape of filtered dataset: {df_filtered.shape}")

# 2. Compute mean (μ) and standard deviation (σ) for each feature per class
stats = df_filtered.groupby('target').agg(['mean', 'std']).T
print("\nMean and Std Dev table for all 4 features:")
print(stats)

# 3. Using the formula to compute Fisher Score for each feature
# Formula: Fisher Score = (μ1 - μ2)² / (σ1² + σ2²)
fisher_scores = {}

for feature in iris.feature_names:
    mu0 = stats.loc[(feature, 'mean'), 0]
    mu1 = stats.loc[(feature, 'mean'), 1]
    sigma0 = stats.loc[(feature, 'std'), 0]
    sigma1 = stats.loc[(feature, 'std'), 1]
    
    score = ((mu0 - mu1)**2) / (sigma0**2 + sigma1**2)
    fisher_scores[feature] = score

fisher_series = pd.Series(fisher_scores)
print("\nFisher Scores:")
print(fisher_series)

# 4. Sort the features in descending order
ranked_features = fisher_series.sort_values(ascending=False)
print("\nRanked List of Features:")
print(ranked_features)

best_feature = ranked_features.index[0]
worst_feature = ranked_features.index[-1]
print(f"\nBest feature: {best_feature}")
print(f"Worst feature: {worst_feature}")

# 5. Plot a bar chart and explanation
plt.figure(figsize=(10, 6))
ranked_features.plot(kind='bar', color='skyblue', edgecolor='black')
plt.title('Fisher Scores for Iris Features (Setosa vs Versicolor)')
plt.xlabel('Features')
plt.ylabel('Fisher Score')
plt.xticks(rotation=45)
plt.grid(axis='y', linestyle='--', alpha=0.7)
plt.show()

# Conclusion
print("\nObservation:")
print("A high Fisher Score indicates that the feature has a large difference between class means ")
print("and low intra-class variance, making it an excellent discriminator for classification.")
